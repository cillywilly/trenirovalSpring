DROP TABLE IF EXISTS training;

CREATE TABLE `TRAINING` (
                            `ID` INTEGER NOT NULL  AUTO_INCREMENT,
                            `TRAINING_TOPIC` VARCHAR(50) NULL  DEFAULT NULL,
                            `DATE` VARCHAR(50) NULL DEFAULT NULL,
                            `USER_ID` INTEGER NULL DEFAULT NULL,
                            PRIMARY KEY (`ID`)
);

DROP TABLE IF EXISTS exercise;

CREATE TABLE `EXERCISE` (
                            `ID` INTEGER NOT NULL  AUTO_INCREMENT,
                            `EXERCISE_NAME` VARCHAR(50) NULL DEFAULT NULL,
                            PRIMARY KEY (`ID`)
);

DROP TABLE IF EXISTS APPROACH;

CREATE TABLE `APPROACH` (
                            `ID` INTEGER NOT NULL  AUTO_INCREMENT,
                            `TRAINING_ID` INTEGER NULL DEFAULT NULL,
                            `EXERCISE_ID` INTEGER NULL DEFAULT NULL,
                            `APPROACH_NUMBER` INTEGER NULL DEFAULT NULL,
                            `REPEAT` INTEGER NULL DEFAULT NULL,
                            `WORKLOAD` INTEGER NULL DEFAULT NULL,
                            PRIMARY KEY (`ID`)
);

DROP TABLE IF EXISTS user;

CREATE TABLE `USER` (
                        `ID` INTEGER NOT NULL AUTO_INCREMENT,
                        `LOGIN` VARCHAR(50) NOT NULL,
                        `PASSWORD` VARCHAR(50) NOT NULL,
                        PRIMARY KEY (`ID`)
);

ALTER TABLE training ADD FOREIGN KEY (USER_ID) REFERENCES user (`ID`);
ALTER TABLE APPROACH ADD FOREIGN KEY (TRAINING_ID) REFERENCES training (`ID`);
ALTER TABLE APPROACH ADD FOREIGN KEY (EXERCISE_ID) REFERENCES exercise (`ID`);
